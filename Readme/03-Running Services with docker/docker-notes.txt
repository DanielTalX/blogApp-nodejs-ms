- Build an image based on the dockerfile in the current directory.  Tag it as 'mydockerid/posts'
docker build -t mydockerid/posts .

- Create and start a container based on the provided image id or tag
docker run mydockerid/posts

- Create and start container, but also override the default command
docker run -it [image id or image tag] [cmd]
==> docker run -it mydockerid/posts sh

- Print out information about all of the running containers
docker ps

- Execute the given command in a running container
docker exec -it [container_id] [cmd]
==> docker run -it 4sfg5651sd2 sh

- Print out logs from the given container
docker logs [container_id]

- push docker image to dockerhub
docker push mydockerid/posts
